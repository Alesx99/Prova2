<script setup>
import { ref, computed } from 'vue'

const count = ref(0)

Pinia - State Management

Pinia è il nuovo store management per Vue 3:

Esempio: Store Pinia

import { defineStore } from 'pinia'

export const useUserStore = defineStore('user', {
  state: () => ({
    user: null,
    isAuthenticated: false
  }),
  
  actions: {
    login(userData) {
      this.user = userData
      this.isAuthenticated = true
    },
    
    logout() {
      this.user = null
      this.isAuthenticated = false
    }
  }
})

Backend & API

REST APIs

Le REST APIs sono fondamentali per applicazioni moderne:

Esempio: API Endpoint

// GET /api/users
app.get('/api/users', async (req, res) => {
  try {
    const users = await User.find()
    res.json(users)
  } catch (error) {
    res.status(500).json({ error: error.message })
  }
})

// POST /api/users
app.post('/api/users', async (req, res) => {
  try {
    const user = new User(req.body)
    await user.save()
    res.status(201).json(user)
  } catch (error) {
    res.status(400).json({ error: error.message })
  }
})

Autenticazione JWT

JSON Web Tokens per autenticazione sicura:

Esempio: Middleware JWT

const jwt = require('jsonwebtoken')

const authenticateToken = (req, res, next) => {
  const authHeader = req.headers['authorization']
  const token = authHeader && authHeader.split(' ')[1]
  
  if (!token) {
    return res.status(401).json({ error: 'Token richiesto' })
  }
  
  jwt.verify(token, process.env.JWT_SECRET, (err, user) => {
    if (err) return res.status(403).json({ error: 'Token non valido' })
    req.user = user
    next()
  })
}

Database

Modelli Dati

Design di database efficienti e scalabili:

Esempio: Schema Utente

const userSchema = new mongoose.Schema({
  username: {
    type: String,
    required: true,
    unique: true,
    minlength: 3
  },
  email: {
    type: String,
    required: true,
    unique: true,
    match: /^\S+@\S+\.\S+$/
  },
  password: {
    type: String,
    required: true,
    minlength: 6
  },
  role: {
    type: String,
    enum: ['user', 'admin'],
    default: 'user'
  },
  createdAt: {
    type: Date,
    default: Date.now
  }
})

Testing

Unit Testing

Test unitari per garantire la qualità del codice:

Esempio: Test Unitario

import { describe, it, expect } from 'vitest'
import { mount } from '@vue/test-utils'
import Counter from './Counter.vue'

describe('Counter', () => {

Integration Testing

Test di integrazione per verificare il funzionamento complessivo:

Esempio: Test API

import request from 'supertest'
import app from '../app'

describe('API Users', () => {
  it('GET /api/users restituisce lista utenti', async () => {
    const response = await request(app)
      .get('/api/users')
      .expect(200)
    
    expect(Array.isArray(response.body)).toBe(true)
  })
  
  it('POST /api/users crea nuovo utente', async () => {
    const userData = {
      username: 'testuser',
      email: 'test@example.com',
      password: 'password123'
    }
    
    const response = await request(app)
      .post('/api/users')
      .send(userData)
      .expect(201)
    
    expect(response.body.username).toBe(userData.username)
  })
})

DevOps

CI/CD Pipeline

Pipeline di integrazione e deployment continui:

Esempio: GitHub Actions

name: CI/CD Pipeline

on:
  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v4
    - uses: actions/setup-node@v4
      with:
        node-version: '18'
    - run: npm ci
    - run: npm test
    - run: npm run build

  deploy:
    needs: test
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    steps:
    - uses: actions/checkout@v4
    - name: Deploy to production
      run: echo "Deploying to production..."

Best Practices

Clean Code

Principi per codice pulito e manutenibile:

  • Nomi significativi - Variabili e funzioni con nomi chiari
  • Funzioni piccole - Una responsabilità per funzione
  • Commenti utili - Spiegare il "perché", non il "cosa"
  • DRY - Don't Repeat Yourself

Performance

Ottimizzazioni per performance elevate:

  • Lazy Loading - Caricamento on-demand dei componenti
  • Code Splitting - Divisione del bundle in chunks
  • Caching - Cache intelligente per dati e asset
  • Compressione - Gzip/Brotli per ridurre dimensioni

Sicurezza

Best practices per applicazioni sicure:

  • Input Validation - Validazione di tutti gli input
  • HTTPS - Crittografia end-to-end
  • Rate Limiting - Protezione da attacchi DDoS
  • SQL Injection - Uso di prepared statements
